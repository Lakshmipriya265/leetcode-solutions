class Solution:
    def generateParenthesis(self, n):
      result = []
      def brackets(current, open_bracket, close_bracket):
       if len(current) == 2*n:
        result.append(current)
        return
       if open_bracket < n:
        brackets(current +"(", open_bracket+1, close_bracket)
       if close_bracket < open_bracket:
        brackets(current + ")", open_bracket, close_bracket+1)
      brackets("", 0, 0)
      return result
