class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int h) {
        int left = 1;
        int right = *max_element(piles.begin(), piles.end());
        while(left<right)
        {
            int speed = (left+right)/2;
            int hs=0;
            for(auto i:piles)
            hs += i/speed + (i%speed != 0);
            if(hs<=h)
            right = speed;
            else
            left = speed+1;
        }
        return left;
    }
};
